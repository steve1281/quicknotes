
set ECHO OFF
set SERVEROUTPUT on
declare

  function myFunction (myValue in number, myFactor in number) return number is
  myFactor_to_use number;
  minimum_wage number := 5;
  begin
    if (myFactor is null) or (myFactor < minimum_wage) then
      myFactor_to_use := minimum_wage;
    else
        myFactor_to_use := myFactor;
    end if;
    
    return myValue * myFactor_to_use;
  end;
        

BEGIN
  
  dbms_output.enable(1000);
  dbms_output.put_line(myFunction(40,10));
  dbms_output.put_line(myFunction(40,2));
  dbms_output.put_line(myFunction(40,null));
END ;
