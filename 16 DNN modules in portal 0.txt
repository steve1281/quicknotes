-- list of all modules/definitions/source used in myTbayTel (LIVE)
-- as of Sept 9. 2011
--
-- svf
--
-- note: I prebuilt the module list using other queries.
---
select * from modules m

inner join ModuleDefinitions md on m.ModuleDefID=md.ModuleDefID
inner join ModuleControls mc on md.ModuleDefID = mc.ModuleDefID
where m.portalid = 0
