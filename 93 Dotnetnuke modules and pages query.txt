--SELECT * 
SELECT md.*, m.*, tm.*, t.*
FROM dbo.TabModules tm
JOIN dbo.Tabs t ON tm.TabID = t.TabID
JOIN dbo.Modules m ON m.ModuleID = tm.ModuleID
JOIN dbo.ModuleDefinitions md ON md.ModuleDefID = m.ModuleDefID
WHERE t.tabid IN (
SELECT tabid FROM dbo.Tabs 
WHERE TabName='MyTbaytel'
OR ParentID = 93)
