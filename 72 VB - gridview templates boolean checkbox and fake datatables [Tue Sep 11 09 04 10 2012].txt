
Templates for boolean checkbox things


Create a gridview with checkbox code  (like so:

                 <asp:TemplateField HeaderText="M">
                      <ItemTemplate>
                          <asp:CheckBox ID="chkM" runat="server" Checked='<%# Bind("M") %>'></asp:CheckBox>
                      </ItemTemplate>
                  </asp:TemplateField>
)

In the code behind you will need a DataTable with a "M" in it.

Protected Sub FakeFeatureList()
        Dim dt As DataTable = New DataTable   
	'' create a fake table.

        dt.Columns.Add(New DataColumn("M"))    '' <------ our boolean
        dt.Columns.Add(New DataColumn("R"))
        dt.Columns.Add(New DataColumn("SelectedFeature"))
        dt.Columns.Add(New DataColumn("Amount"))
        dt.Columns.Add(New DataColumn("From"))
        dt.Columns.Add(New DataColumn("Through"))
        dt.Columns.Add(New DataColumn("BilledThrough"))
        dt.Columns.Add(New DataColumn("BackThrough"))
        dt.Columns.Add(New DataColumn("BackBillCR"))
        dt.Columns.Add(New DataColumn("Pct"))
        dt.Columns.Add(New DataColumn("Qty"))

	'' create a row
	Dim row As DataRow = dt.NewRow()
        row.ItemArray = New Object() {
            True,				'' <------ our boolean
            True,
            "GSM Transition 1",
            "20.0",
            "11/15/2010",
            "",
            "08/09/2012",
            True,
            "",
            "1"
        }
	
	'' add it
        dt.Rows.Add(row)

	'' source and bind.
        grdFeatures.DataSource = dt
        grdFeatures.DataBind()

    End Sub
