--NOTE: change the last line from ROLLBACK to COMMIT to actually make the username change
BEGIN TRAN

DECLARE @old VARCHAR(MAX)
DECLARE @new VARCHAR(MAX)

SET @old = 'old_name'
SET @new = 'new_name'

DECLARE @cnt INT

SELECT  @cnt = COUNT(*)
FROM    dbo.users
WHERE   username LIKE @new

IF @cnt > 0 
    BEGIN
    PRINT 'Username ' + @new + ' already in use'

        RETURN

    END

UPDATE TOP ( 1 )
        users
SET     username = @new
WHERE   username = @old

UPDATE TOP ( 1 )
        aspnet_Users
SET     username = @new ,
        loweredusername = LOWER(@new)
WHERE   username = @old
