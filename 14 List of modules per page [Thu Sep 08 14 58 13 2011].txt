
Run this from HOST->SQL

/****** Script for SelectTopNRows command from SSMS  ******/
SELECT TOP 1000 [TabID]
      ,[TabOrder]
      ,[PortalID]
      ,[TabName]
      ,[IsVisible]
      ,[ParentId]
      ,[Level]
      ,[IconFile]
      ,[DisableLink]
      ,[Title]
      ,[Description]
      ,[KeyWords]
      ,[IsDeleted]
      ,[Url]
      ,[SkinSrc]
      ,[ContainerSrc]
      ,[TabPath]
      ,[StartDate]
      ,[EndDate]
      ,[RefreshInterval]
      ,[PageHeadText]
      ,[IsSecure]
      ,[PermanentRedirect]
  FROM [DotNetNuke].[dbo].[Tabs]
  where PortalID=0
  order by TabName
