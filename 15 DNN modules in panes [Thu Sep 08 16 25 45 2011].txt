SELECT TOP 1000 [TabModuleID]
      ,tm.[TabID]
      ,t.TabName
      ,tm.[ModuleID]
      ,m.ModuleTitle
      ,[PaneName]
      ,[ModuleOrder]
      ,[CacheTime]
      ,[Alignment]
      ,[Color]
      ,[Border]
      ,tm.[IconFile]
      ,[Visibility]
      ,tm.[ContainerSrc]
      ,[DisplayTitle]
      ,[DisplayPrint]
      ,[DisplaySyndicate]
   FROM [DotNetNuke].[dbo].[TabModules] tm
  INNER JOIN [DotNetNuke].[dbo].[Tabs] t on t.TabID = tm.TabID
  INNER JOIN [DotNetNuke].[dbo].[Modules] m on tm.ModuleID=m.ModuleID
  where t.PortalID=0 and
  t.TabID=36
