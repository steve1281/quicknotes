declare @temp table (
	hse_no    varChar(10),
	hse_sufx  varChar(10),
	st_nm     varChar(100),
	st_sufx   varChar(10),
	dirl_cd   varChar(10),
	com_nm    varChar(100)
	)

insert @temp 
select
	hse_no 
	,hse_sufx
	,st_nm
	,st_sufx
	,dirl_cd
	,com_nm
FROM 
	OPENQUERY(MARTENS,
		N'
		select
			Distinct
			hse_no
		   ,hse_sufx
		   ,st_nm
		   ,st_sufx
		   ,dirl_cd
		   ,com_nm
		from
		vmr013eqp_adrs_ap
		')


select count(st_nm) , st_nm from @temp group by st_nm
